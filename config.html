config.rb

Compass Configuration

Setting css, sass, images, javascripts and fonts directories inside app. Read more about different options on Compass Configuration references

http_path = "/" css_dir = "app/css" sass_dir = "app/scss" images_dir = "app/img" javascripts_dir = "app/js" fonts_dir = "app/css/fonts"

Output style :nested,:expanded,:compact, or compressed

output_style = :nested environment = :stand_alone relative_assets = true

No line comments

line_comments = false color_output = false

Preferred syntax scss

preferred_syntax = :scss
